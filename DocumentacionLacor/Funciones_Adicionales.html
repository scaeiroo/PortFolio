Funci√≥n de Configuraci√≥n

1.1 CargarConfiguracion
La funci√≥n CargarConfiguracion() carga y filtra las configuraciones de una base de datos en formato JSON para ajustarlas a las preferencias de base de datos, alias y dominio. La configuraci√≥n se almacena globalmente para ser utilizada en otros scripts.

CargarConfiguracion <- function(FicheroConf = './Config/Config.json', TipoDBElegido ='SQL Server', AliasElegido = 'DBA', DominioElegido = 'Remoto'){
  .GlobalEnv$configuracion <- fromJSON(FicheroConf)
  .GlobalEnv$configuracion <- .GlobalEnv$configuracion %>% filter((toupper(TipoDB) == toupper(TipoDBElegido))) 
  .GlobalEnv$configuracion <- .GlobalEnv$configuracion %>% filter((toupper(Alias) == toupper(AliasElegido))) 
  .GlobalEnv$configuracion <- .GlobalEnv$configuracion %>% filter((toupper(Dominio) == toupper(DominioElegido)))
  
  return(.GlobalEnv$configuracion)
}

Par√°metros:

  • FicheroConf: Ruta al archivo de configuraci√≥n (por defecto ‚Äò./Config/Config.json‚Äô).

  • TipoDBElegido, AliasElegido, DominioElegido: Par√°metros para filtrar las configuraciones.

GuardarCSV <- function(df, nombre_archivo, ruta) {
  tryCatch({
    if (!dir.exists(ruta)) {
      dir.create(ruta, recursive = TRUE) 
    }
    ruta_completa <- file.path(ruta, paste0(nombre_archivo, ".csv"))
    write.csv(df, ruta_completa, row.names = FALSE, fileEncoding = "UTF-8", na = "")
    cat("‚úÖ Guardado correctamente:", ruta_completa, "- Registros:", nrow(df), "\n")
  }, error = function(e) {
    cat("‚ùå Error al guardar:", nombre_archivo, "- Mensaje:", e$message, "\n")
  })
}

Par√°metros:

  • df: El DataFrame a guardar.

  • nombre_archivo: El nombre del archivo.

  • ruta: La ruta donde guardar el archivo.

GuardarRData <- function(objeto, ruta_archivo) {
  tryCatch({
    nombre_objeto <- deparse(substitute(objeto))
    assign(nombre_objeto, objeto)
    save(list = nombre_objeto, file = ruta_archivo)
    if (file.exists(ruta_archivo)) {
      registros <- if (is.data.frame(objeto)) nrow(objeto) else "N/A"
      cat("‚úÖ Archivo guardado: '", ruta_archivo, "' - Objeto: ", nombre_objeto, " - Registros: ", registros, "\n")
    } else {
      cat("‚ùå Error: No se pudo guardar '", ruta_archivo, "'.\n")
    }
  }, error = function(e) {
    cat("‚ùå Error al guardar '", ruta_archivo, "': ", e$message, "\n")
  })
}

  • objeto: El objeto R que deseas guardar (como un DataFrame, lista, etc.).

CargarCSV <- function(ruta_archivo) {
  if (!file.exists(ruta_archivo)) {
    cat("‚ùå El archivo CSV '", ruta_archivo, "' no se encuentra. Verifica el nombre o la fecha.\n")
    stop("Archivo CSV no encontrado.")
  }
  
  tryCatch({
    df <- read.csv(ruta_archivo, sep = ",", stringsAsFactors = FALSE, fileEncoding = "UTF-8")
    if (ncol(df) == 1) {
      df <- read.csv(ruta_archivo, sep = ";", stringsAsFactors = FALSE, fileEncoding = "UTF-8")
    }
    cat("‚úÖ El archivo CSV '", ruta_archivo, "' se carg√≥ correctamente. Registros: ", nrow(df), " - Columnas: ", ncol(df), "\n")
    return(df)
  }, error = function(e) {
    cat("‚ùå Error al cargar '", ruta_archivo, "': ", e$message, "\n")
    stop(e)
  })
}

Par√°metros:

  • ruta_archivo: La ruta completa del archivo CSV que deseas cargar.

Funci√≥n Obtener √öltima Fecha de Actualizaci√≥n

2.2 ObtenerUltimaFechaAct
Esta funci√≥n ObtenerUltimaFechaAct() revisa los archivos de producci√≥n diaria almacenados en la carpeta de salida (./Output/) y devuelve la fecha m√°s reciente de los archivos guardados. Es √∫til para conocer la √∫ltima actualizaci√≥n de datos disponible en el sistema.

ObtenerUltimaFechaAct <- function() {
  archivo_total <- './Output/Produccion_Diario_TOTAL.RData'
  
  if (!file.exists(archivo_total)) {
    return(NULL) 
  }
  
  ProduccionDiario_TOTAL_DF <- get(load(archivo_total))
  
  if (nrow(ProduccionDiario_TOTAL_DF) > 0 && "Jornada" %in% colnames(ProduccionDiario_TOTAL_DF)) {
    return(max(ProduccionDiario_TOTAL_DF$Jornada, na.rm = TRUE))
  } else {
    warning("El archivo total est√° vac√≠o o no contiene la columna 'Jornada'.")
    return(NULL)
  }
}

Funci√≥n Filtrar Duplicados

2.3 FiltrarDuplicadosTOTAL
La funci√≥n FiltrarDuplicadosTOTAL() elimina registros duplicados o no v√°lidos del conjunto de datos de producci√≥n, priorizando registros con incidencias, fechas de fin m√°s recientes y mayores cantidades buenas.

FiltrarDuplicadosTOTAL <- function(df) {
  if (!dir.exists("./Output")) {
    dir.create("./Output")
  }
  df <- df %>%
    filter(!(Estado == "PARO" & is.na(Operario) & is.na(EXPEDIENTE) & is.na(REFERENCIA) & is.na(OF)))
  
  df <- df %>%
    group_by(OF, FASE, EXPEDIENTE, REFERENCIA, MAQUINA, FechaInicio_pr, FechaFin_pr) %>%
    filter(!(is.na(MATERIAL_SALIDA) | MATERIAL_SALIDA == "")) %>%
    ungroup()
  
  df_filtrado <- df %>%
    group_by(OF, FechaInicio_pr, Estado, FASE, MATERIAL_SALIDA) %>% 
    mutate(IncidenciaPrioridad = !is.na(Incidencia),
           OperarioSeleccionado = coalesce(first(Operario), NA)) %>%  
    slice_max(order_by = IncidenciaPrioridad, with_ties = TRUE) %>% 
    slice_max(order_by = FechaFin_pr, with_ties = TRUE) %>%
    slice_max(order_by = Cantidad_Buenas, with_ties = FALSE) %>%
    select(-IncidenciaPrioridad, -OperarioSeleccionado) %>%
    ungroup()
  
  filas_eliminadas <- nrow(df) - nrow(df_filtrado)
  
  cat("üîÑ Total de filas originales: ", nrow(df), "\n")
  cat("üéâ Total de filas despu√©s del filtrado: ", nrow(df_filtrado), "\n")
  cat("üî• Filas eliminadas: ", filas_eliminadas, "\n")
  
  return(df_filtrado)
}

Par√°metros:

  • df: El DataFrame del que deseas eliminar los duplicados.

Funci√≥n Filtrar Duplicados TOTAL

2.3 FiltrarDuplicadosTOTALDIARIO
La funci√≥n FiltrarDuplicadosTOTALDIARIO() optimiza el proceso de filtrado de duplicados al trabajar solo con un subconjunto de los datos totales, enfoc√°ndose en los registros recientes para mejorar el rendimiento.

FiltrarDuplicadosTOTALDIARIO <- function(df_total, df_diario) {
  n_filas_a_registrar <- nrow(df_diario) * 2
  filas_a_registrar <- tail(df_total, n_filas_a_registrar)
  filas_a_registrar <- filas_a_registrar %>%
    filter(!(Estado == "PARO" & is.na(Operario) & is.na(EXPEDIENTE) & is.na(REFERENCIA) & is.na(OF)))
  
  filas_a_registrar <- filas_a_registrar %>%
    group_by(OF, FASE, EXPEDIENTE, REFERENCIA, MAQUINA, FechaInicio_pr, FechaFin_pr) %>%
    filter(!(is.na(MATERIAL_SALIDA) | MATERIAL_SALIDA == "")) %>%
    ungroup()
  
  df_filtrado <- filas_a_registrar %>%
    group_by(OF, FechaInicio_pr, Estado, FASE, MATERIAL_SALIDA) %>% 
    mutate(IncidenciaPrioridad = !is.na(Incidencia),
           OperarioSeleccionado = coalesce(first(Operario), NA)) %>%  
    slice_max(order_by = IncidenciaPrioridad, with_ties = TRUE) %>% 
    slice_max(order_by = FechaFin_pr, with_ties = TRUE) %>%
    slice_max(order_by = Cantidad_Buenas, with_ties = FALSE) %>%
    select(-IncidenciaPrioridad, -OperarioSeleccionado) %>%
    ungroup()
  
  df_sin_filas_a_registrar <- df_total %>%
    anti_join(filas_a_registrar, by = c("OF", "FechaInicio_pr", "Estado", "FASE", "MATERIAL_SALIDA"))
  df_actualizado <- bind_rows(df_sin_filas_a_registrar, df_filtrado)
  filas_eliminadas <- nrow(df_total) - nrow(df_actualizado)
  
  cat("üîÑ Total de filas originales: ", nrow(df_total), "\n")
  cat("üéâ Total de filas despu√©s del filtrado: ", nrow(df_actualizado), "\n")
  cat("üî• Filas eliminadas o reemplazadas: ", filas_eliminadas, "\n")
  
  
  return(df_actualizado)
}

Par√°metros:

  • df_total: El DataFrame total que contiene todos los registros hist√≥ricos
  • df_diario: El DataFrame con los nuevos registros diarios
